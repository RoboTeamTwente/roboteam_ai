33 /* The ParamEvent Anchor determines whether a ParamEvent is
34  * tied to the start time and value, end time and value, or
35  * the full Range. This determines how the start and end
36  * values and times will change (or remain fixed) when other
37  * ParaEvents are added.
38  */
39 enum class Anchor { NONE = 0x0, START = 0x1, END = 0x2, ALL = 0x3 };
40 inline Anchor operator&(Anchor a, Anchor b) {
41  using T = std::underlying_type<Anchor>::type;
42  return static_cast<Anchor>(static_cast<T>(a) & static_cast<T>(b));
43 }
