339 static void jcv_edge_create(jcv_edge* e, jcv_site* s1, jcv_site* s2) {
340  e->next = 0;
341  e->sites[0] = s1;
342  e->sites[1] = s2;
343  e->pos[0].x = JCV_INVALID_VALUE;
344  e->pos[0].y = JCV_INVALID_VALUE;
345  e->pos[1].x = JCV_INVALID_VALUE;
346  e->pos[1].y = JCV_INVALID_VALUE;
347 
348  // Create line equation between S1 and S2:
349  // jcv_real a = -1 * (s2->p.y - s1->p.y);
350  // jcv_real b = s2->p.x - s1->p.x;
351  // //jcv_real c = -1 * (s2->p.x - s1->p.x) * s1->p.y + (s2->p.y - s1->p.y) * s1->p.x;
352  //
353  // // create perpendicular line
354  // jcv_real pa = b;
355  // jcv_real pb = -a;
356  // //jcv_real pc = pa * s1->p.x + pb * s1->p.y;
357  //
358  // // Move to the mid point
359  // jcv_real mx = s1->p.x + dx * jcv_real(0.5);
360  // jcv_real my = s1->p.y + dy * jcv_real(0.5);
361  // jcv_real pc = ( pa * mx + pb * my );
362 
363  jcv_real dx = s2->p.x - s1->p.x;
364  jcv_real dy = s2->p.y - s1->p.y;
365  int dx_is_larger = (dx * dx) > (dy * dy); // instead of fabs
366 
367  // Simplify it, using dx and dy
368  e->c = dx * (s1->p.x + dx * (jcv_real)0.5) + dy * (s1->p.y + dy * (jcv_real)0.5);
369 
370  if (dx_is_larger) {
371  e->a = (jcv_real)1;
372  e->b = dy / dx;
373  e->c /= dx;
374  } else {
375  e->a = dx / dy;
376  e->b = (jcv_real)1;
377  e->c /= dy;
378  }
379 }

412  y1 = pymax;

533 static int jcv_halfedge_rightof(const jcv_halfedge* he, const jcv_point* p) {
534  const jcv_edge* e = he->edge;
535  const jcv_site* topsite = e->sites[1];
536 
537  int right_of_site = (p->x > topsite->p.x) ? 1 : 0;
538  if (right_of_site && he->direction == JCV_DIRECTION_LEFT) return 1;
539  if (!right_of_site && he->direction == JCV_DIRECTION_RIGHT) return 0;
540 
541  jcv_real dxp, dyp, dxs, t1, t2, t3, yl;
542 
543  int above;
544  if (e->a == (jcv_real)1) {
545  dyp = p->y - topsite->p.y;
546  dxp = p->x - topsite->p.x;
547  int fast = 0;
548  if ((!right_of_site & (e->b < (jcv_real)0)) | (right_of_site & (e->b >= (jcv_real)0))) {
549  above = dyp >= e->b * dxp;
550  fast = above;
551  } else {
552  above = (p->x + p->y * e->b) > e->c;
553  if (e->b < (jcv_real)0) above = !above;
554  if (!above) fast = 1;
555  }
556  if (!fast) {
557  dxs = topsite->p.x - e->sites[0]->p.x;
558  above = e->b * (dxp * dxp - dyp * dyp) < dxs * dyp * ((jcv_real)1 + (jcv_real)2 * dxp / dxs + e->b * e->b);
559  if (e->b < (jcv_real)0) above = !above;
560  }
561  } else // e->b == 1
562  {
563  yl = e->c - e->a * p->x;
564  t1 = p->y - yl;
565  t2 = p->x - topsite->p.x;
566  t3 = yl - topsite->p.y;
567  above = t1 * t1 > (t2 * t2 + t3 * t3);
568  }
569  return (he->direction == JCV_DIRECTION_LEFT ? above : !above);
570 }

588  const jcv_halfedge* he;

591  e = e1;
592  } else {
593  he = he2;
594  e = e2;
595  }
596 
597  int right_of_site = out->x >= e->sites[1]->p.x;
598  if ((right_of_site && he->direction == JCV_DIRECTION_LEFT) || (!right_of_site && he->direction == JCV_DIRECTION_RIGHT)) {
599  return 0;
600  }
601 
602  return 1;
603 }

607 static int jcv_pq_moveup(jcv_priorityqueue* pq, int pos) {
608  jcv_halfedge** items = (jcv_halfedge**)pq->items;
609  jcv_halfedge* node = items[pos];
610 
611  for (int parent = (pos >> 1); pos > 1 && jcv_halfedge_compare(items[parent], node); pos = parent, parent = parent >> 1) {
612  items[pos] = items[parent];
613  items[pos]->pqpos = pos;
614  }
615 
616  node->pqpos = pos;
617  items[pos] = node;
618  return pos;
619 }

624  jcv_halfedge** items = (jcv_halfedge**)pq->items;

630  jcv_halfedge** items = (jcv_halfedge**)pq->items;
631  jcv_halfedge* node = items[pos];
632

637  pos = child;
638  child = jcv_pq_maxchild(pq, pos);
639  }
640 
641  items[pos] = node;
642  items[pos]->pqpos = pos;
643  return pos;
644 }

762 static inline jcv_real jcv_calc_sort_metric(const jcv_site* site, const jcv_graphedge* edge) {
763  // We take the average of the two points, since we can better distinguish between very small edges
764  jcv_real half = 1 / (jcv_real)2;
765  jcv_real x = (edge->pos[0].x + edge->pos[1].x) * half;
766  jcv_real y = (edge->pos[0].y + edge->pos[1].y) * half;
767  jcv_real diffy = y - site->p.y;
768  jcv_real angle = JCV_ATAN2(diffy, x - site->p.x);
769  if (diffy < 0) angle = angle + 2 * JCV_PI;
770  return (jcv_real)angle;
771 }

789 static void jcv_finishline(jcv_context_internal* internal, jcv_edge* e) {
790  if (!jcv_edge_clipline(internal, e)) {
791  return;
792  }
793 
794  // Make sure the graph edges are CCW
795  int flip = jcv_determinant(&e->sites[0]->p, &e->pos[0], &e->pos[1]) > (jcv_real)0 ? 0 : 1;
796 
797  for (int i = 0; i < 2; ++i) {

799 
800  ge->edge = e;
801  ge->next = 0;
802  ge->neighbor = e->sites[1 - i];
803  ge->pos[flip] = e->pos[i];
804  ge->pos[1 - flip] = e->pos[1 - i];
805  ge->angle = jcv_calc_sort_metric(e->sites[i], ge);
806 
807  jcv_sortedges_insert(e->sites[i], ge);
808 
809  // check that we didn't accidentally add a duplicate (rare), then remove it
810  if (ge->next && ge->angle == ge->next->angle) {
811  if (jcv_point_eq(&ge->pos[0], &ge->next->pos[0]) && jcv_point_eq(&ge->pos[1], &ge->next->pos[1])) {
812  ge->next = ge->next->next; // Throw it away, they're so few anyways
813  }
814  }
815  }

859 void inline jcv_boxshape_fillgaps(const jcv_clipper* clipper, jcv_context_internal* allocator, jcv_site* site) {
860  // They're sorted CCW, so if the current->pos[1] != next->pos[0], then we have a gap
861  jcv_graphedge* current = site->edges;
862  if (!current) {
863  // No edges, then it should be a single cell
864  assert(allocator->numsites == 1);
865 
866  jcv_graphedge* gap = jcv_alloc_graphedge(allocator);
867  gap->neighbor = 0;
868  gap->pos[0] = clipper->min;
869  gap->pos[1].x = clipper->max.x;
870  gap->pos[1].y = clipper->min.y;
871  gap->angle = jcv_calc_sort_metric(site, gap);
872  gap->next = 0;
873  gap->edge = jcv_create_gap_edge(allocator, site, gap);
874 
875  current = gap;
876  site->edges = gap;
877  }
878 
879  jcv_graphedge* next = current->next;
880  if (!next) {
881  // Only one edge, then we assume it's a corner gap
882  jcv_graphedge* gap = jcv_alloc_graphedge(allocator);
883  jcv_create_corner_edge(allocator, site, current, gap);
884  gap->edge = jcv_create_gap_edge(allocator, site, gap);
885 
886  gap->next = current->next;
887  current->next = gap;
888  current = gap;
889  next = site->edges;
890  }
891 
892  while (current && next) {
893  if (jcv_point_on_box_edge(&current->pos[1], &clipper->min, &clipper->max) && !jcv_point_eq(&current->pos[1], &next->pos[0])) {
894  // Border gap
895  if (current->pos[1].x == next->pos[0].x || current->pos[1].y == next->pos[0].y) {
896  jcv_graphedge* gap = jcv_alloc_graphedge(allocator);
897  gap->neighbor = 0;
898  gap->pos[0] = current->pos[1];
899  gap->pos[1] = next->pos[0];
900  gap->angle = jcv_calc_sort_metric(site, gap);
901  gap->edge = jcv_create_gap_edge(allocator, site, gap);
902 
903  gap->next = current->next;
904  current->next = gap;
905  } else if (jcv_point_on_box_edge(&current->pos[1], &clipper->min, &clipper->max) && jcv_point_on_box_edge(&next->pos[0], &clipper->min, &clipper->max)) {
906  jcv_graphedge* gap = jcv_alloc_graphedge(allocator);
907  jcv_create_corner_edge(allocator, site, current, gap);
908  gap->edge = jcv_create_gap_edge(allocator, site, gap);
909  gap->next = current->next;
910  current->next = gap;
911  } else {
912  // something went wrong, abort instead of looping indefinitely
913  break;
914  }
915  }
916 
917  current = current->next;
918  if (current) {
919  next = current->next;
920  if (!next) next = site->edges;
921  }
922  }
923 }
